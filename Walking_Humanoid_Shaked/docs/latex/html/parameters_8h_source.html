331 void supplementDefaults()
332 {
333 if(lambda < 2)
334 lambda = 4 + (int) (3.0*log((double) N));
335 if(mu <= 0)
336 mu = lambda / 2;
337 if(!weights)
338 setWeights(weightMode);
339
340 if(cs > 0)
341 cs *= (mueff + 2.) / (N + mueff + 3.);
342 if(cs <= 0 || cs >= 1)
343 cs = (mueff + 2.) / (N + mueff + 3.);
344
345 if(ccumcov <= 0 || ccumcov > 1)
346 ccumcov = 4. / (N + 4);
347
348 if(mucov < 1)
349 mucov = mueff;
350 T t1 = 2. / ((N + 1.4142)*(N + 1.4142));
351 T t2 = (2.* mueff - 1.) / ((N + 2.)*(N + 2.) + mueff);
352 t2 = (t2 > 1) ? 1 : t2;
353 t2 = (1. / mucov)* t1 + (1. - 1. / mucov)* t2;
354 if(ccov >= 0)
355 ccov *= t2;
356 if(ccov < 0 || ccov > 1)
357 ccov = t2;
358
359 if(diagonalCov < 0)
360 diagonalCov = 2 + 100. * N / sqrt((double) lambda);
361
362 if(stopMaxFunEvals <= 0)
363 stopMaxFunEvals = facmaxeval * 900 * (N + 3)*(N + 3);
364 else
365 stopMaxFunEvals *= facmaxeval;
366
367 if(stopMaxIter <= 0)
368 stopMaxIter = ceil((double) (stopMaxFunEvals / lambda));
369
370 if(damps < T(0))
371 damps = T(1);
372 damps = damps
373 * (T(1) + T(2)*(std::max)(T(0), std::sqrt((mueff - T(1)) / (N + T(1))) - T(1)))
374 * (T) (std::max)(T(0.3), T(1) - // modify for short runs
375 (T) N / (T(1e-6) + (std::min)(stopMaxIter, stopMaxFunEvals / lambda)))
376 + cs;
377
378 if(updateCmode.modulo < 0)
379 updateCmode.modulo = 1. / ccov / (double) N / 10.;
380 updateCmode.modulo *= facupdateCmode;
381 if(updateCmode.maxtime < 0)
382 updateCmode.maxtime = 0.20; // maximal 20% of CPU-time
383 }

388 void setWeights(Weights mode)
389 {
390 if(weights)
391 delete[] weights;
392 weights = new T[mu];
393 switch(mode)
394 {
395 case LINEAR_WEIGHTS:
396 for(int i = 0; i < mu; ++i) weights[i] = mu - i;
397 break;
398 case EQUAL_WEIGHTS:
399 for(int i = 0; i < mu; ++i) weights[i] = 1;
400 break;
401 case LOG_WEIGHTS:
402 default:
403 for(int i = 0; i < mu; ++i) weights[i] = log(mu + 1.) - log(i + 1.);
404 break;
405 }
406
407 // normalize weights vector and set mueff
408 T s1 = 0, s2 = 0;
409 for(int i = 0; i < mu; ++i)
410 {
411 s1 += weights[i];
412 s2 += weights[i]*weights[i];
413 }
414 mueff = s1*s1/s2;
415 for(int i = 0; i < mu; ++i)
416 weights[i] /= s1;
417
418 if(mu < 1 || mu > lambda || (mu == lambda && weights[0] == weights[mu - 1]))
419 throw std::runtime_error("setWeights(): invalid setting of mu or lambda");
420 }
